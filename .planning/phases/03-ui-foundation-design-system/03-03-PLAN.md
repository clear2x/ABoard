     fn paste_to_active(content: String, app: tauri::AppHandle) -> Result<(), String> {
         // Write content to clipboard
         use tauri::Manager;
         let clipboard = app.state::<tauri_plugin_clipboard_manager::ClipboardManager>();
         // Use clipboard manager to write text, then simulate Cmd+V
         // Write to clipboard first
         let _ = clipboard.write_text(&content);

         // For now: use CGEvent on macOS to simulate paste

             use core_graphics::event_source::CGEventSource;
             use core_graphics::event_source::CGEventSourceStateID::HIDSystemState;
             let source = CGEventSource::new(HIDSystemState).map_err(|e| e.to_string())?;
             // Cmd down
             let cmd_down = CGEvent::new_keyboard_event(&source, 55, true).map_err(|e| e.to_string())?;
             cmd_down.set_flags(CGEventFlags::CGEventFlagCommand);
             let _ = cmd_down.post(core_graphics::event::CGEventTapLocation::HID);
             // V down
             let v_down = CGEvent::new_keyboard_event(&source, 9, true).map_err(|e| e.to_string())?;
             v_down.set_flags(CGEventFlags::CGEventFlagCommand);
             let _ = v_down.post(core_graphics::event::CGEventTapLocation::HID);
             // V up
             let v_up = CGEvent::new_keyboard_event(&source, 9, false).map_err(|e| e.to_string())?;

         }
         Ok(())
     }

1. Create `src/components/FloatingPopup.tsx`:
   ```typescript
   import { createSignal, onMount, onCleanup, For, Show } from "solid-js";
   import { invoke } from "@tauri-apps/api/core";
   import { getCurrentWindow } from "@tauri-apps/api/window";
   import { items, loadHistory, type ClipboardItem } from "../stores/clipboard";

   export default function FloatingPopup() {
     const [selectedIndex, setSelectedIndex] = createSignal(0);
     const [popupItems, setPopupItems] = createSignal<ClipboardItem[]>([]);

     onMount(async () => {
       await loadHistory(0, 10); // Load recent 10 items
       setPopupItems(items().slice(0, 10));

       // Focus the popup window for keyboard events
       await getCurrentWindow().setFocus();

       // Listen for keyboard events on the document
       const handler = (e: KeyboardEvent) => {
         const current = selectedIndex();
         const list = popupItems();
         if (e.key === "ArrowDown") {
           e.preventDefault();
           setSelectedIndex(Math.min(current + 1, list.length - 1));
         } else if (e.key === "ArrowUp") {
           e.preventDefault();
           setSelectedIndex(Math.max(current - 1, 0));
         } else if (e.key === "Enter") {
           e.preventDefault();
           selectAndPaste(list[current]);
         } else if (e.key === "Escape") {
           e.preventDefault();
           getCurrentWindow().hide();
         }
       };
       document.addEventListener("keydown", handler);
       onCleanup(() => document.removeEventListener("keydown", handler));
     });

     async function selectAndPaste(item: ClipboardItem) {
       if (!item) return;
       try {
         await invoke("paste_to_active", { content: item.content });
       } catch (e) {
         console.error("[FloatingPopup] Paste failed:", e);
       }
       await getCurrentWindow().hide();
     }

     return (
       <div class="window-bg min-h-screen text-white flex flex-col p-3 rounded-xl select-none">
         <div class="text-xs text-[var(--color-text-muted)] mb-2 px-1">
           Quick Paste
         </div>
         <div class="flex flex-col gap-1 overflow-y-auto flex-1">
           <Show when={popupItems().length === 0}>
             <div class="text-center text-[var(--color-text-muted)] py-8 text-sm">
               No clipboard items
             </div>
           </Show>
           <For each={popupItems()}>
             {(item, index) => (
               <div
                 class={`p-2 rounded-[var(--radius-md)] cursor-pointer transition-smooth ${
                   index() === selectedIndex()
                     ? "bg-[var(--color-accent)]/20 border border-[var(--color-accent)]/50"
                     : "hover:bg-[var(--color-bg-card-hover)] border border-transparent"
                 }`}
                 onClick={() => { setSelectedIndex(index()); selectAndPaste(item); }}
               >
                 <div class="flex items-center gap-2">
                   <Show when={item.pinned}>
                     <span class="w-1.5 h-1.5 rounded-full bg-yellow-400 shrink-0" />
                   </Show>
                   <span class="text-sm text-[var(--color-text-secondary)] truncate">
                     {item.type === "image" ? "[Image]" : item.content.slice(0, 80)}
                   </span>
                 </div>
                 <div class="text-[10px] text-[var(--color-text-muted)] mt-0.5">
                   {new Date(item.timestamp).toLocaleTimeString()}
                 </div>
               </div>
             )}
           </For>
         </div>
         <div class="text-[10px] text-[var(--color-text-muted)] mt-2 text-center">
           Up/Down to navigate - Enter to paste - Esc to close
         </div>
       </div>
     );
   }
   ```
