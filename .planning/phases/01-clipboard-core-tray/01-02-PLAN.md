## Key Tauri v2 API Patterns

### System Tray (Rust)
```rust
use tauri::{
    menu::{Menu, MenuItem},
    tray::TrayIconBuilder,
    Manager,
};

let show_i = MenuItem::with_id(app, "show", "Show Window", true, None::<&str>)?;
let pause_i = MenuItem::with_id(app, "pause", "Pause Monitoring", true, None::<&str>)?;
let quit_i = MenuItem::with_id(app, "quit", "Quit", true, None::<&str>)?;
let menu = Menu::with_items(app, &[&show_i, &pause_i, &quit_i])?;

let tray = TrayIconBuilder::new()
    .icon(app.default_window_icon().unwrap().clone())
    .menu(&menu)
    .menu_on_left_click(true)
    .on_menu_event(move |app, event| match event.id().as_ref() {
        "show" => { /* show window */ },
        "pause" => { /* toggle monitoring */ },
        "quit" => { app.exit(0); },
        _ => (),
    })
    .build(app)?;
```

### Global Shortcut (Rust)
```rust
use tauri_plugin_global_shortcut::{Code, GlobalShortcutExt, Modifiers, Shortcut, ShortcutState};

let shortcut = Shortcut::new(
    Some(Modifiers::SUPER | Modifiers::SHIFT),
    Code::KeyV
);
app.handle().plugin(
    tauri_plugin_global_shortcut::Builder::new()
        .with_handler(move |app, _shortcut, event| {
            if event.state() == ShortcutState::Pressed {
                // Toggle window visibility
            }
        })
        .build(),
)?;
app.global_shortcut().register(shortcut)?;
```

### Window Close to Tray (Frontend)
```typescript
import { getCurrentWindow } from "@tauri-apps/api/window";
await getCurrentWindow().onCloseRequested(async (event) => {
    event.preventDefault();
    await getCurrentWindow().hide();
});
```

### Window Position at Cursor
```typescript
import { getCurrentWindow } from "@tauri-apps/api/window";
import { cursorPosition } from "@tauri-apps/api/window";
// Get cursor position, calculate center offset, set window position
const pos = await cursorPosition();
const windowSize = await getCurrentWindow().innerSize();
await getCurrentWindow().setPosition(new LogicalPosition(
    pos.x - windowSize.width / 2,
    pos.y - windowSize.height / 2
));
```

1. **Create `src-tauri/src/tray.rs`** with the system tray setup:

   ```rust
   use tauri::{
       AppHandle,
       Manager,
       menu::{Menu, MenuItem},
       tray::{TrayIconBuilder, TrayIconEvent, MouseButton, MouseButtonState},
   };

   pub fn setup_tray(app: &AppHandle) -> Result<(), Box<dyn std::error::Error>> {
       let show_i = MenuItem::with_id(app, "show", "Show Window", true, None::<&str>)?;
       let pause_i = MenuItem::with_id(app, "pause", "Pause Monitoring", true, None::<&str>)?;
       let quit_i = MenuItem::with_id(app, "quit", "Quit", true, None::<&str>)?;

       let menu = Menu::with_items(app, &[&show_i, &pause_i, &quit_i])?;

       let _tray = TrayIconBuilder::new()
           .icon(app.default_window_icon().unwrap().clone())
           .menu(&menu)
           .menu_on_left_click(true)
           .tooltip("ABoard - Clipboard Manager")
           .on_menu_event(move |app, event| match event.id().as_ref() {
               "show" => {
                   if let Some(webview_window) = app.get_webview_window("main") {
                       let _ = webview_window.show();
                       let _ = webview_window.set_focus();
                   }
               }
               "pause" => {
                   // Call the toggle_monitoring from clipboard module
                   let paused = crate::clipboard::toggle_monitoring();
                   let pause_menu_item = app.menu().find("pause").unwrap();
                   // Update menu text based on state
                   // This requires casting to MenuItem -- use the MenuItem::set_text method
                   if let Some(item) = pause_menu_item.as_menuitem() {
                       let _ = item.set_text(if paused { "Resume Monitoring" } else { "Pause Monitoring" });
                   }
               }
               "quit" => {
                   app.exit(0);
               }
               _ => (),
           })
           .on_tray_icon_event(|tray, event| {
               if let TrayIconEvent::Click {
                   button: MouseButton::Left,
                   button_state: MouseButtonState::Up,
                   ..
               } = event {
                   let app = tray.app_handle();
                   if let Some(webview_window) = app.get_webview_window("main") {
                       let _ = webview_window.show();
                       let _ = webview_window.set_focus();
                   }
               }
           })
           .build(app)?;

       Ok(())
   }
   ```

   IMPORTANT: Adapt the above to match the actual Tauri v2 API. The `MenuItem::find`, `as_menuitem`, and `set_text` APIs may have slightly different signatures in the actual crate version. Read the crate docs or source if compilation fails.
