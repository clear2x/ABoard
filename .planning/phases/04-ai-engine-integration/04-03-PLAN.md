```typescript
// Plan 02
invoke('ai_download_model', { url: string, name: string }): Promise<ModelInfo>
invoke('ai_delete_model', { modelId: string }): Promise<void>
invoke('ai_get_config'): Promise<AiConfig>
invoke('ai_set_config', { config: AiConfig }): Promise<void>
```

TypeScript 类型 (在前端定义):
```typescript
interface InferenceRequest {
  prompt: string;
  system_prompt?: string;
  max_tokens?: number;
  temperature?: number;
  top_p?: number;
}

interface ModelInfo {
  id: string;
  name: string;
  filename: string;
  file_size: number;
  status: string;        // "available" | "downloading" | "error"
  downloaded_at: number;
  is_active: boolean;
}

interface AiConfig {
  active_provider: string;  // "Local" | "OpenAi" | "Anthropic"
  model_path?: string;
  context_length: number;
  openai_api_key?: string;
  openai_endpoint: string;
  openai_model: string;
  anthropic_api_key?: string;
  anthropic_model: string;
}
```
