**路由策略（启发式规则）**:

```rust
/// 复杂度评估结果
#[derive(Debug, Clone, PartialEq)]
pub enum Complexity {
    Simple,   // 短文本 (<200 chars)，分类、标签、检测
    Medium,   // 中等文本 (200-1000 chars)，摘要、改写
    Complex,  // 长文本 (>1000 chars)，翻译长文、详细总结、多步推理
}

/// 路由决策
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct RoutingDecision {
    pub provider: ProviderType,
    pub complexity: Complexity,
    pub reason: String,  // 路由原因说明
}
```

**ComplexityRouter 实现**:
```rust
pub struct ComplexityRouter;

impl ComplexityRouter {
    /// 分析 InferenceRequest，返回路由决策
    pub fn route(request: &InferenceRequest, config: &AiConfig) -> RoutingDecision {
        let complexity = Self::assess_complexity(request);

        // 检查本地是否可用
        let local_available = Self::is_local_available(config);
        let cloud_available = Self::is_cloud_available(config);

        match (complexity, local_available, cloud_available) {
            // 简单任务，本地可用 -> 本地
            (Complexity::Simple, true, _) => RoutingDecision { provider: Local, ... "Simple task, routing to local" },
            // 中等任务，本地可用 -> 本地（本地足够处理）
            (Complexity::Medium, true, false) => RoutingDecision { provider: Local, ... "Medium task, no cloud, routing to local" },
            // 中等任务，云端可用 -> 云端（更高质量）
            (Complexity::Medium, _, true) => RoutingDecision { provider: OpenAi, ... "Medium task, routing to cloud for better quality" },
            // 复杂任务，云端可用 -> 云端
            (Complexity::Complex, _, true) => RoutingDecision { provider: OpenAi, ... "Complex task, routing to cloud" },
            // 任何复杂度，仅本地可用 -> 本地
            (_, true, false) => RoutingDecision { provider: Local, ... "Only local available" },
            // 任何复杂度，仅云端可用 -> 云端
            (_, false, true) => RoutingDecision { provider: OpenAi, ... "Only cloud available" },
            // 都不可用 -> Local (将在 infer 时返回错误)
            (_, false, false) => RoutingDecision { provider: Local, ... "No provider available" },
        }
    }

    fn assess_complexity(request: &InferenceRequest) -> Complexity {
        let prompt = &request.prompt;
        let prompt_len = prompt.chars().count();

        // 关键词检测: system_prompt 暗示任务类型
        let sys = request.system_prompt.as_deref().unwrap_or("");

        // 简单任务关键词: classify, tag, detect, categorize, label
        let simple_keywords = ["classify", "tag", "detect", "categorize", "label", "identify", "分类", "标签", "检测"];
        // 复杂任务关键词: translate, summarize, explain, rewrite, format, detailed
        let complex_keywords = ["translate", "summarize", "explain", "rewrite", "format", "detailed", "翻译", "总结", "改写", "详细"];

        let prompt_lower = prompt.to_lowercase();
        let sys_lower = sys.to_lowercase();
        let combined = format!("{} {}", prompt_lower, sys_lower);

        if simple_keywords.iter().any(|k| combined.contains(k)) && prompt_len < 500 {
            Complexity::Simple
        } else if complex_keywords.iter().any(|k| combined.contains(k)) || prompt_len > 1000 {
            Complexity::Complex
        } else if prompt_len < 200 {
            Complexity::Simple
        } else {
            Complexity::Medium
        }
    }

    fn is_local_available(config: &AiConfig) -> bool {
        // 简单检查: model_path 有值
        config.model_path.is_some()
    }

    fn is_cloud_available(config: &AiConfig) -> bool {
        config.openai_api_key.is_some() || config.anthropic_api_key.is_some()
    }
}
```

```rust
#[tauri::command]
pub async fn ai_infer_auto(
    state: tauri::State<'_, AiState>,
    request: InferenceRequest,
) -> Result<InferenceAutoResponse, String> {
    // 1. 验证 prompt 长度 (复用 MAX_PROMPT_LEN)
    if request.prompt.len() > MAX_PROMPT_LEN {
        return Err(format!("Prompt too long (max {} bytes)", MAX_PROMPT_LEN));
    }

    // 2. 读取当前配置
    let config = state.config.lock().await.clone();

    // 3. 如果不是 Auto 模式，直接走当前 provider
    if config.active_provider != ProviderType::Auto {
        let provider = state.provider.lock().await;
        let result = tokio::time::timeout(
            std::time::Duration::from_secs(INFERENCE_TIMEOUT_SECS),
            provider.infer(request),
        ).await.map_err(|_| "Inference timed out")??;
        return Ok(InferenceAutoResponse {
            response: result,
            routing_decision: None,
        });
    }

    // 4. Auto 模式: 使用路由器决策
    let decision = router::ComplexityRouter::route(&request, &config);

    // 5. 根据决策选择 provider
    let provider: Box<dyn InferenceProvider> = match decision.provider {
        ProviderType::Local => {
            let local = if let Some(ref path) = config.model_path {
                local::LocalProvider::new(path.clone())
            } else {
                local::LocalProvider::new(String::new())
            };
            Box::new(local)
        }
        ProviderType::OpenAi => {
            let p = cloud::OpenAiProvider::new(
                config.openai_api_key.clone(),
                config.openai_endpoint.clone(),
                config.openai_model.clone(),
            );
            Box::new(p)
        }
        ProviderType::Anthropic => {
            let p = cloud::AnthropicProvider::new(
                config.anthropic_api_key.clone(),
                config.anthropic_model.clone(),
            );
            Box::new(p)
        }
        ProviderType::Auto => unreachable!(), // 已经处理过 Auto
    };

    // 6. 执行推理 (带超时)
    let result = tokio::time::timeout(
        std::time::Duration::from_secs(INFERENCE_TIMEOUT_SECS),
        provider.infer(request),
    ).await.map_err(|_| "Inference timed out")??;

    Ok(InferenceAutoResponse {
        response: result,
        routing_decision: Some(decision),
    })
}
```
