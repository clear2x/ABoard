   ```typescript
   export async function loadHistory(offset: number = 0, limit: number = 50) {
     setLoading(true);
     try {
       const result = await invoke<ClipboardItem[]>("get_history", { offset, limit });
       setItems(result);
     } catch (e) {
       console.error("[store] Failed to load history:", e);
     } finally {
       setLoading(false);
     }
   }
   ```

5. **searchHistory function** — called by SearchBar:
   ```typescript
   export async function searchHistory(query: string) {
     if (!query.trim()) {
       await loadHistory();
       return;
     }
     setLoading(true);
     try {
       const result = await invoke<ClipboardItem[]>("search_history", { query, offset: 0, limit: 50 });
       setItems(result);
     } catch (e) {
       console.error("[store] Search failed:", e);
     } finally {
       setLoading(false);
     }
   }
   ```

6. **deleteItems function**:
   ```typescript
   export async function deleteItems(ids: string[]) {
     try {
       await invoke("delete_items", { ids });
       await loadHistory(); // Refresh list after delete
     } catch (e) {
       console.error("[store] Delete failed:", e);
     }
   }
   ```

7. **pinItem / unpinItem functions**:
   ```typescript
   export async function pinItem(id: string) {
     try {
       await invoke("pin_item", { id });
       await loadHistory();
     } catch (e) {
       console.error("[store] Pin failed:", e);
     }
   }

   export async function unpinItem(id: string) {
     try {
       await invoke("unpin_item", { id });
       await loadHistory();
     } catch (e) {
       console.error("[store] Unpin failed:", e);
     }
   }
   ```

   ```typescript
       return [item, ...prev];
     });
   }
   ```

9. **Keep startClipboardListener and stopClipboardListener** unchanged — they still listen to "clipboard-update" events and call addItem.

10. **Remove MAX_ITEMS constant** — items are now persisted, no need for in-memory cap.
